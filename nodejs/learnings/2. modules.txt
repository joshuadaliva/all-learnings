----------commonJS

function add(x, y){
    return x + y;
}

function subtract(x, y){
    return x - y;
}


// 1. exporting a single value / object
module.exports = add
const sum = require("./app.js")
sum.add()

// 2. exportin multiple values / object

module.exports = {
    add,
    subtract
}

const {add, subtract} = require("./app");
// or
const math = require("./app")




-------------- es module


// enabling es module by .mjs file or setting the type inside the package.json as type:moduke

// one by one export
export function add(x , y){
    return x + y;
}

export function subtract(x , y){
    return x + y;
}

// multiple
export {add, subtract}


// importing
import {add, subtract} from "./index"
